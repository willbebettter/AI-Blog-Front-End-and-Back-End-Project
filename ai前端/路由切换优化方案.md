/**
 * 前端路由切换优化方案
 * 
 * 问题分析：
 * 1. 路由切换时组件重新加载导致延迟
 * 2. 数据获取请求过多
 * 3. 动画时间过长
 * 
 * 优化措施：
 * 1. 优化路由切换逻辑
 * 2. 减少不必要的 API 请求
 * 3. 优化动画时间
 * 4. 添加预加载机制
 */

// 1. 优化 App.vue 中的路由切换逻辑
// 修改 watch 监听器，只在必要时重新加载数据
watch(
  () => route.path,
  (newPath, oldPath) => {
    // 只在切换到用户中心时才加载数据
    if (newPath === '/user-center' && oldPath !== '/user-center') {
      // 显示 loading
      isLoading.value = true
      
      // 预加载：在显示 loading 后立即开始获取数据
      setTimeout(async () => {
        await fetchUserInfo()
        isLoading.value = false
      }, 100) // 短暂延迟，让用户看到 loading
    }
  },
  { immediate: true }
)

// 2. 优化 UserCenter.vue 中的数据获取逻辑
// 修改 fetchUserInfo 函数，减少不必要的请求
const fetchUserInfo = async () => {
  // 1. 先尝试从缓存获取
  if (getCachedData()) {
    console.log('使用缓存的用户数据')
    return
  }
  
  try {
    // 2. 并行获取用户基本信息和博客数据
    const [userInfoResponse, allBlogsResponse] = await Promise.allSettled([
      // 从 API 获取用户信息
      fetch('/api/user/info').then(res => res.json()).catch(() => ({
        id: 1,
        nickname: '测试用户',
        email: 'test@example.com',
        avatar: '',
        articleCount: 5,
        likeCount: 12,
        collectCount: 8,
        followerCount: 20,
        followingCount: 15
      })),
      // 只获取点赞和收藏的博客，而不是全部博客
      fetch(`/api/blogs?ids=${likedIds.join(',')}`).then(res => res.json()).catch(() => [])
    ]);

    // 处理用户信息
    if (userInfoResponse.status === 'fulfilled') {
      dataCache.value.user = userInfoResponse.value;
    }

    // 3. 优化：只获取特定 ID 的博客
    if (likedIds.length > 0 || collectedIds.length > 0) {
      try {
        const filteredBlogResponse = await fetch(`/api/blogs?ids=${likedIds.join(',')}`);
        if (filteredBlogResponse.ok) {
          const filteredBlogs = await filteredBlogResponse.json();
          const blogMap = new Map(filteredBlogs.map(blog => [blog.id, blog]));

          dataCache.value.likedBlogs = likedIds.map(id => blogMap.get(id)).filter(Boolean);
          dataCache.value.collectedBlogs = collectedIds.map(id => blogMap.get(id)).filter(Boolean);
        }
      } catch (filterError) {
        console.error('获取特定博客失败，使用后备方案:', filterError);
        // 后备方案：获取全部博客后过滤
        const allBlogs = allBlogsResponse.status === 'fulfilled' ? allBlogsResponse.value : [];
        dataCache.value.likedBlogs = allBlogs.filter(blog => likedIds.includes(blog.id));
        dataCache.value.collectedBlogs = allBlogs.filter(blog => collectedIds.includes(blog.id));
      }
    }

    // 4. 设置关注者信息
    dataCache.value.following = [
      {
        id: 2,
        nickname: '作者 1',
        avatar: '',
        articleCount: 10
      },
      {
        id: 3,
        nickname: '作者 2',
        avatar: '',
        articleCount: 8
      }
    ];

    // 5. 保存缓存
    saveCachedData(dataCache.value);

  } catch (error) {
    console.error('获取用户信息失败:', error);
    // 设置默认值避免界面空白
    dataCache.value.user = {
      id: null,
      nickname: '未知用户',
      email: '',
      avatar: '',
      articleCount: 0,
      likeCount: 0,
      collectCount: 0,
      followerCount: 0,
      followingCount: 0
    };
    dataCache.value.likedBlogs = [];
    dataCache.value.collectedBlogs = [];
    dataCache.value.following = [];
  }
};

// 3. 优化后端 API 响应速度
// 在 BlogController 中添加批量查询优化
@GetMapping("/blogs")
public ResponseEntity<ApiResponse<List<Blog>>> getBlogs(
        @RequestParam(required = false) List<Long> ids,
        @RequestParam(required = false) String title,
        @RequestParam(required = false) String category,
        @RequestParam(required = false) String tag,
        @RequestParam(defaultValue = "0") int page,
        @RequestParam(defaultValue = "10") int size) {
    
    // 优化：如果只查询特定 ID 的博客，使用 IN 查询
    if (ids != null && !ids.isEmpty()) {
        try {
            // 使用批量查询，而不是逐个查询
            List<Blog> blogs = blogRepository.findByIdIn(ids);
            return ResponseEntity.ok(new ApiResponse<>(blogs));
        } catch (Exception e) {
            // 如果批量查询失败，返回空列表
            return ResponseEntity.ok(new ApiResponse<>(Collections.emptyList()));
        }
    }
    
    // 其他查询逻辑...
}

// 4. 添加预加载机制
// 在 App.vue 中添加预加载逻辑
onMounted(() => {
  // 预加载常用数据
  if (userStore.isLoggedIn) {
    // 预加载用户中心数据
    fetchUserInfo()
  }
})

// 5. 优化动画时间
// 在 App.vue 中修改 transition 配置
transition 
  name="fade" 
  mode="out-in"
  appear
  appear-from="opacity-0"
  appear-to="opacity-100"
  appear-transition="opacity 0.1s ease"  // 缩短动画时间
/>

// 6. 添加数据预取
// 在路由守卫中添加预取逻辑
router.beforeEach((to, from, next) => {
  // 如果是用户中心页面，预取数据
  if (to.path === '/user-center') {
    // 预取数据
    fetchUserInfo()
  }
  
  next()
})
